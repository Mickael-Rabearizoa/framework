cree une fonction cast(Object a_caster , Class classe)
cette fonction va caster l'objet "a_caster" en une instance de "classe", la mettre dans une variable "result" de type Object
cette fonction va ensuite retourner   "result" 



public static Object cast(Object a_caster, Class classe) throws IllegalAccessException, InstantiationException {
        Object result = classe.newInstance();
        result = classe.cast(a_caster);
        return result;
    }
